# 混合App 升级 🚀

TIP

UniApp 升级注意事项 （本功能仅限于静态文件传输）

  1. 前端H5 页面中 创建 静态JSON 文件， 内容填写版本号即可，其他信息根据项目需求添加
  2. App 打包时需配置好版本信息，版本信息要与 H5 静态JSON中的版本内容一致
  3. 下载成功后会自动弹出升级提示

# onLoad 页面首次打开调用

onLoad() {
    // 获取客户端信息
    uni.getSystemInfo({
        success: (res) => {
            //检测当前平台，如果是安卓则启动安卓更新  
            if (res.platform == "android") {
                // 升级函数
                this.AndroidCheckUpdate()
            }
        }
    })
}

# methods 函数

methods: {
    // App 升级函数
    AndroidCheckUpdate: function() {
        var vm = this;
        // 获取版本信息
        plus.runtime.getProperty(plus.runtime.appid, (info) => {
            vm.version = info.version;
            // 自己请求自己，对应H5本地JSON 文件所写内容

            // {
            //     "name": "应用名称",
            //     "version": "1.0.1",
            //     "versionCode": "101"
            // }

            uni.request({
                url: '/static/version.json',
                success: res => {
                    // 对比版本信息  —— 上方data 中自行定义  **version**
                    if (res.data.version > vm.version) {
                        // 判断当前网络环境
                        if (plus.networkinfo.getCurrentType() != 3) {
                            uni.showToast({
                                title: '当前环境非WIFI，不更新，仅作提示',
                                mask: false,
                                duration: 5000,
                                icon: "none"
                            });
                            return;
                        }
                        //提醒用户更新 —— 禁用掉取消按钮，使用户强行更新
                        uni.showModal({
                            title: "系统提示",
                            content: 'WIFI环境 更新程序提示...',
                            confirmText: "立即升级",
                            showCancel: false,
                            success: (res) => {
                                if (res.confirm) {
                                    // 调用下载函数
                                    vm.Downloadprogress()
                                }
                            }
                        })
                    }
                }
            });
        })
    },
    // 触发下载并添加下载进度显示
    Downloadprogress() {
        // 纯前端更新，采用nginx 映射本地资源文件进行下载，根据需求可填写其他下载连接
        // plus.downloader.createDownload(url, {}, function(){})
        var dtask = plus.downloader.createDownload( "http://XXX.XXXXX.XXX:xxx/apk/XXX.apk", {}, function(d, status) {
                // 下载完成  
                if (status == 200) {
                    plus.runtime.install(plus.io.convertLocalFileSystemURL(d.filename), {}, {}, function( error) {
                        uni.showToast({
                            title: '安装失败',
                            mask: false,
                            duration: 1500
                        });
                    })
                } else {
                    uni.showToast({
                        title: '更新失败',
                        mask: false,
                        duration: 1500
                    });
                }
            });
        try {
            dtask.start(); // 开启下载的任务
            var prg = 0;
            var showLoading = plus.nativeUI.showWaiting("正在下载"); //创建一个showWaiting对象

            dtask.addEventListener('statechanged', function(task, status) {
                console.log('更新测试：', task, status)
                // 给下载任务设置一个监听 并根据状态  做操作
                switch (task.state) {
                    case 1:
                        showLoading.setTitle("正在下载");
                        break;
                    case 2:
                        showLoading.setTitle("已连接到服务器");
                        break;
                    case 3:
                        prg = parseInt((parseFloat(task.downloadedSize) / parseFloat(task.totalSize)) *
                            100);
                        showLoading.setTitle("正在下载,请勿操作 " + prg + "%  ");
                        break;
                    case 4:
                        plus.nativeUI.closeWaiting();
                        //下载完成
                        break;
                }
            });
        } catch (err) {
            console.log('错误信息', err)
            plus.nativeUI.closeWaiting();
        }
    }
}
