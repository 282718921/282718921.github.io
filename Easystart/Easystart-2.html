# UNIAPP精确定位(百度地图) 📍

TIP

因项目需求使用的是百度地图（bd09）本方法是使用高德地图（gcj02）获取定位在通过转换坐标系完成

  1. 该项目为 UniApp(App) 通过 webview 嵌套 UniApp(H5) 完成的混合App (百度地图Api 无法在uniapp 的App 板块中使用)
  2. 首先在H5 项目中引入百度地图Api
  3. 引入成功后加载地图模块
  4. 调用App 中的定位Api获取当前位置，在通过webview 的message 方法进行 App端 与 H5端的通信
  5. H5端获取到定位信息通过坐标系转换得到百度地图坐标系的定位点
  6. 将转换后的定位点信息设置到地图中心点
  7. 自行添加img图片并将图片设置为垂直水平居中(仅作用于展示，实际定位点为地图的中心点)
  8. 拖拽地图获取地图中心点信息并传递至表单页面

# 获取定位信息并传递到H5地图页面完成地图渲染

onLoad() {
    const vm = this
    // 获取定位信息
    this.getGpsData()
},
methods:{
    // 初始化地图
    initMap() {
        const vm = this
        vm.BDmap = new BMap.Map('container'); // 创建Map实例
        // vm.BDmap.centerAndZoom(new BMap.Point(116.404, 39.915), 18); // 初始化地图,设置中心点坐标和地图级别
        vm.BDmap.enableScrollWheelZoom(true); // 开启鼠标滚轮缩放
        // 添加切换图层控件到地图左下角
        var mapType = new BMap.MapTypeControl({
            "mapTypes": [BMAP_NORMAL_MAP, BMAP_HYBRID_MAP],
            "anchor": BMAP_ANCHOR_BOTTOM_LEFT
        });
        vm.BDmap.addControl(mapType); //2D图，卫星图
        // 设置中心点 vm.MapCenter 在获取定位信息中赋值
        if (window.plus) {
            setTimeout(function() {
                var point = new BMap.Point(vm.MapCenter.lng, vm.MapCenter.lat);
                vm.BDmap.centerAndZoom(point, 18);
            }, 500)
        }
        // 监听地图拖拽 将获取到的中心点位信息赋值到 MapCenter 对象中
        vm.BDmap.addEventListener("dragend", function() {
            //获取地图中心点经纬度坐标
            var center = vm.BDmap.getCenter();
            console.log("地图中心点变更为：" + center.lng + ", " + center.lat);
            vm.MapCenter.lng = center.lng
            vm.MapCenter.lat = center.lat
            //坐标点添加到地图
            // var point = new BMap.Point(center.lng, center.lat);
            // var marker = new BMap.Marker(point);
            // vm.BDmap.addOverlay(marker);
        });
    },
    // 获取经纬度信息
    getGpsData() {
        const vm = this
        if (window.plus) {
            uni.showLoading({
                title: '定位中...',
                mask: true
            });
            // 调用App 中的方法
            uni.webView.postMessage({
                data: {
                    'type': 'getLocation', // App中用于判断方法的标识
                    "data": '' // H5 传递信息到 App
                }
            });
            // 接受App壳子给的参数
            window.getLocation = function(data) {
                console.log('getGpsData==========>>>>>>>>>', JSON.stringify(data))
                // 进行坐标系转换
                var gcj02tobd09 = coordtransform.gcj02tobd09(data.longitude, data.latitude);
                vm.MapCenter.lat = gcj02tobd09[1]
                vm.MapCenter.lng = gcj02tobd09[0]
                setTimeout(function() {
                    uni.hideLoading();
                }, 2000);
            }
        } else {
            Toast('访问设备暂不支持定位...')
        }
    },
}

# 壳子定义message方法提供通讯

methods: {
    // 定义通讯方法
    message(data) {
        const vm = this
        // 扫码
        if (data.detail.data[0].type === 'SaoMa') {
            uni.navigateTo({
                url: '../saoma/text?text=' + this.text + '&type=' + this.type
            });
        }
        // 过去定位信息
        if (data.detail.data[0].type === 'getLocation') {
            const vm = this
            vm.getLocation()


        }
    },
    // 获取经纬度信息结果
    getLocation() {
        uni.getLocation({
            type: 'gcj02',
            success: function(res) {
                const latitude = res.latitude;
                const longitude = res.longitude;
                const data = {
                    latitude: "",
                    longitude: ""
                }
                data.latitude = latitude
                data.longitude = longitude
                // 将数据传递至 H5 项目中
                const funName = 'getLocation'
                const currentWebview = vm.$scope.$getAppWebview().children()[0];
                currentWebview.evalJS(`${funName}(${JSON.stringify(data)})`);
            }
        });
    },
}
