//  App 端 数据处理

message(data) {
    const vm = this
    // 扫码
    if (data.detail.data[0].type === 'SaoMa') {
        uni.navigateTo({
            url: '../saoma/text?text=' + this.text + '&type=' + this.type
        });
    }
    // 选择地图
    if (data.detail.data[0].type === 'Maps') {
        console.log('===========', data.detail.data[0].data)
        uni.navigateTo({
            url: '../maps/index?data=' + data.detail.data[0].data
        });
    }
    // 新版本选择地图
    if (data.detail.data[0].type === 'NewMapS') {
        const vm = this
        uni.getLocation({
            type: 'gcj02', //返回可以用于uni.openLocation的经纬度
            success: function(res) {
                const latitude = res.latitude;
                const longitude = res.longitude;
                const data = {
                    latitude: "",
                    longitude: ""
                }
                data.latitude = latitude
                data.longitude = longitude
                const funName = 'NewMapS'
                const currentWebview = vm.$scope.$getAppWebview().children()[0];
                currentWebview.evalJS(`${funName}(${JSON.stringify(data)})`);
            }
        });


    }
},
